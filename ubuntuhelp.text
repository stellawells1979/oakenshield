git branch		查看分支名

git commit -m "首次提交"		推送前先执行此代码，提交到本地仓库

git push -u origin <分支名>		推送到远程仓库

git push -u origin master --force	强制推送到远程仓库

git remote -v		查看当前远程仓库地址
